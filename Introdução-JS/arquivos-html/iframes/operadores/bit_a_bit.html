console.log(5 & 3)   // 1 (0101 & 0011 = 0001) 

console.log(5 | 3)   // 7 (0101 | 0011 = 0111) 

console.log(5 ^ 3)   // 6 (0101 ^ 0011 = 0110) 

console.log(~5)      // -6 (inverte os bits de 5) 

console.log(5 << 1)  // 10 (0101 << 1 = 1010) 

console.log(5 >> 1)  // 2 (0101 >> 1 = 0010) 

console.log(5 >>> 1) // 2 (0101 >>> 1 = 0010)
