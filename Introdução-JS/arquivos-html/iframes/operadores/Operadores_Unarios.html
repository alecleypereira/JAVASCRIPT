let x = 5 
console.log(typeof x)  // 'number' 
delete x               // Remove a variável x 
console.log(void 0)    // undefined 
console.log(+ '10')    // 10 (string para número) 
console.log(- '10')    // -10 (string para número e negação) 
console.log(++x)       // 6 (incrementa x) 
console.log(--x)       // 5 (decrementa x)
