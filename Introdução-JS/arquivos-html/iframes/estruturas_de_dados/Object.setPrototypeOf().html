let heranca = { tipo: "herdado", valor: 42 };
    let pessoa = { nome: "João", idade: 30 };

    // Configurando o protótipo de 'pessoa' para ser 'heranca'
    Object.setPrototypeOf(pessoa, heranca);

    console.log(pessoa.tipo); // "herdado" (propriedade herdada de 'heranca')
    console.log(pessoa.valor); // 42 (também herdada de 'heranca')
    console.log(pessoa.nome);  // "João" (propriedade própria de 'pessoa')
